// reservation.h
#ifndef RESERVATION_H
#define RESERVATION_H

#define ROWS 5
#define COLS 4

// Enum for bus type
typedef enum { AC, NON_AC, SLEEPER } BusType;

// Passenger structure
typedef struct {
    char name[50];
    int age;
    int seatRow;
    int seatCol;
    BusType type;
} Passenger;

// Global variables
extern int seats[ROWS][COLS];
extern Passenger passengers[ROWS*COLS];
extern int passengerCount;

// Function prototypes
void displaySeats();
void bookSeat();
void cancelSeat();

#endif
